📚 INTEGRATION EXAMPLE
═══════════════════════════════════════════════════════════════════════════════
document.addEventListener('DOMContentLoaded', async () => {
  // 1. Setup - load all your app's locations
  const allLocations = await getAllLocations();
  window.nearbyAttractionsFinder.setAppLocations(allLocations);
  // 2. Auto-load when coordinates change
  document.getElementById('latitude').addEventListener('change', async () => {
    const lat = parseFloat(document.getElementById('latitude').value);
    const lng = parseFloat(document.getElementById('longitude').value);
    const name = document.getElementById('locationName').value;
    await autoLoadNearbyAttractions(lat, lng, name, 'attractionsContainer');
  });
  // 3. Handle user clicks
  window.addEventListener('navigate-location', (e) => {
    // User clicked existing location
    navigateToLocationCard(e.detail.location);
  });
  window.addEventListener('add-attraction', (e) => {
    // User clicked "Add to App"
    openAddLocationForm({
      name: e.detail.attraction.name,
      latitude: e.detail.attraction.latitude,
      longitude: e.detail.attraction.longitude,
      type: e.detail.attraction.type,
      description: e.detail.attraction.description,
      googleUrl: e.detail.attraction.googleUrl
    });
  });
});
